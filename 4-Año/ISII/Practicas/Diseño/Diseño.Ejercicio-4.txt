module      ControlBrillo
import      Imagen
exportproc  increase(i Imagen): Imagen
            isMax(i Imagen): Bool
            decrease(i Imagen): Imagen
            isMin(i Imagen): Bool
comments    Funciones basicas para controlar el brillo de
            una imagen. Las funciones isMax y isMin devuelven
            True si el brillo esta al maximo o minimimo
            respectivamente, sino False.
